 15 ( function( $ )
 16 {
 17 	var defaults = {
 18 			modal: "div",
 19 			modalID: "modal",
 20 			buttonClose: ".bt_close:first",
 21 			buttonMini: "button.bt_mini",
 22 			buttonMiniFunction: null,
 23 			create: true,
 24 			remove: true,
 25 			center: false,
 26 			closeOnClick: false,
 27 			closeCallback: null,
 28 			effectVelocity: $.browser.msie ? 0 : 700 ,
 29 			beginShow: true
 30 		},
 31 		queue = [], toback =  [];
 32 	
 33 	$.fn.modal = function( settings, callback )
 34 	{
 35 		typeof settings == "function"
 36 			? callback = settings
 37 			: callback = callback || function(){};
 38 		
 39 		return this.each( function()
 40 		{
 41 			new $m( this, settings, callback );
 42 		});
 43 	}
 44 	
 45 	$.modalFunc = function( container, settings, callback )
 46 	{
 47 		this.settings = $.extend( {}, defaults, settings || {});
 48 		this.div = $( container ).hide();
 49 		this.callback = callback;
 50 		this.clicked = 0;
 51 		
 52 		this.init();
 53 	}
 54 	
 55 	var $m = $.modalFunc;
 56 	
 57 	$m.fn = $m.prototype = {
 58         modal: '0.3'
 59     };
 60 	
 61 	$m.fn.show = function()
 62 	{
 63 		this.div.fadeIn();
 64 		$( ".modalLoader", this.modal ).hide();
 65 	}
 66 		
 67 	$m.fn.toClose = function( callback, mantain )
 68 	{
 69 		var mantain = mantain || false;
 70 		this.close( mantain, callback || function(){} );
 71 	}
 72 	
 73 	$m.fn.extend = $.extend;
 74 	
 75 	$m.fn.extend({
 76 		init: function()
 77 		{
 78 			this.modal = $( this.settings.modal + "#" + this.settings.modalID );
 79 			
 80 			var self = this,
 81 				s = self.settings,
 82 				loader = $( ".modalLoader", this.modal ),
 83 				show = function()
 84 				{
 85 					self.callback.call( self.div, self );
 86 					loader.hide();
 87 				},
 88 				callback = function()
 89 				{
 90 					if( self.settings.create )
 91 					{
 92 						s.beginShow
 93 							? ( s.effectVelocity
 94 								? self.div.fadeIn( show )
 95 								: self.div.show() | show() )
 96 							: self.callback.call( self.div, self );
 97 							
 98 						$.browser.msie && $( "html" ).addClass( "modal" );
 99 						s.center && self.center();
100 					}
101 					else if( !s.create )
102 						self.close();	
103 				}
104 				
105 			this.selects = $( "select:visible" );
106 			
107 			if( !this.modal.length && this.settings.create && this.div.length )
108 			{
109 				this.create();
110 				
111 				s.effectVelocity
112 					? this.modal.fadeIn( this.settings.effectVelocity, callback() )
113 					: this.modal.show() && callback();	
114 			}			
115 			else if( this.modal.length && this.div.length )
116 			{
117 				//self.modal.show()
118 				this.settings.create
119 					? s.effectVelocity 
120 						? self.modal.fadeIn()
121 						: self.modal.show()
122 					: self.modal.hide();
123 					
124 				callback()
125 			}
126 			else if( !this.div.length )
127 				console.error( "The element doesn't were found" );
128 			
129 			$( document ).keyup( function( e )
130 			{
131 				if( self.modal.is( ":visible" ) && self.div.is( ":visible" ) && e.keyCode == 27 )
132 					self.close();
133 			});
134 				
135 			this.settings.closeOnClick && this.modal.unbind().click( function()
136 			{
137 				self.close();
138 				
139 				self.settings.closeCallback && self.settings.closeCallback();
140 			});
141 			
142 			$.browser.msie
143 				&& this.selects.css( "visibility", "hidden" );
144 			
145 			$.browser.msie && $.browser.version.indexOf( "6" ) != -1
146 				&& this.modal.css( "height", $( "html" )[ 0 ].offsetHeight )
147 				&& window.scrollTo( 0, 0 )
148 			
149 			this.buttons();
150 			this.queue();
151 		},
152 		
153 		buttons: function()
154 		{
155 			var self = this
156 				count = 0;
157 			
158 			// 
159 			if( this.settings.buttonClose )
160 			{
161 				var close = this.buttonClose = $( this.settings.buttonClose, this.div );
162 				
163 				close.length && close.click( function( event )
164 				{
165 					!self.clicked && self.close();
166 					!self.clicked && !!self.settings.closeCallback && self.settings.closeCallback();
167 					
168 					return false;
169 				});
170 			}
171 			
172 			// Minimizar
173 			if( this.settings.buttonMini )
174 			{
175 				var mini = this.buttonMini = $( this.settings.buttonMini, this.div );
176 				
177 
178 				mini.length && mini.click( function()
179 				{
180 					if( self.div.is( ":visible" ) || !count )
181 					{
182 						self.settings.buttonMiniFunction && self.settings.buttonMiniFunction();
183 						self.close( true );
184 					}
185 					count++;
186 				});
187 			}
188 		},
189 		
190 		create: function()
191 		{
192 			var s = this.settings;
193 			
194 			this.modal = $( '<' + s.modal + '></' + s.modal + '>' ).attr( "id", s.modalID ).hide().appendTo( "body" );
195 			
196 			this.size();
197 			this.zindex();
198 			
199 			return true;
200 		},
201 		
202 		close: function( mantain, callback )
203 		{
204 			var self = this
205 				callback = callback || function(){},
206 				hide = function()
207 				{
208 					self.settings.effectVelocity > 0 
209 						? self.modal.fadeOut( self.settings.effectVelocity, callback )
210 						: self.modal.hide() && callback();
211 					
212 					self.clicked++;
213 					!mantain && self.settings.remove && self.div.remove() && ( self.clicked = 0 ) ;
214 					
215 					$.browser.msie && $( "html" ).removeClass( "modal" );
216 					self.queue( true );
217 					self.callback;
218 				};
219 			
220 			self.settings.closeCallback && self.settings.closeCallback();
221 			
222 			self.div.is( ":visible" ) && ( $.browser.msie
223 				? self.div.hide() | hide()
224 				: self.div.fadeOut( hide )
225 			)
226 			
227 			$.browser.msie && this.selects.css( "visibility", "visible" ); 
228 		},
229 		
230 		size: function( m )
231 		{
232 			m && ( this.modal = m );
233 			
234 			if( $.browser.msie && $.browser.version == "6.0" )
235 			{
236 				var h = document.body.clientHeight;
237 				var w = document.body.clientWidth;
238 				
239 				//this.modal.css( { "position": "absolute", "height": h + "px", "width": w + "px" } );
240 				this.modal.css( { "position": "absolute", "height": h + "px", "width": "100%" } );
241 			}	
242 			
243 			return true;
244 		},
245 		
246 		center: function()
247 		{
248 			//top = Number( this.div.height() / 2 ) * -1;
249 			
250 			for( name in { "height": 0, "width": 0 } )
251 			{
252 				eval( name + "= Number( this.div." + name + "() / 2 ) * -1" );
253 			}
254 			
255 			this.div.length == 1
256 			   ? this.div.css( { "marginTop": height, "marginLeft": width } )
257 			   : 0;
258 		},
259 		
260 		queue: function( back )
261 		{
262 			if( !back )
263 			{
264 				queue.push( this.settings.modalID );
265 				
266 				if( queue.length > 1 )
267 				{
268 					var id = queue[ queue.length - 2 ],
269 						elem = $( "#" + id + ":visible" );
270 					
271 					elem.length && id != this.settings.modalID && elem.hide() && toback.push( elem );
272 				}	
273 			}
274 			else
275 				toback.length && toback[ toback.length - 1 ].show() && toback.pop() && queue.pop();
276 		},
277 		
278 		zindex: function( zindex )
279 		{
280 			var zindex = zindex ? zindex : ( parseInt( this.div.css( "zIndex" ) ) -1 );
281 			
282 			this.modal.css( "zIndex", zindex );
283 			
284 			return true;
285 		}
286 	});
287 	
288 	$.modal =
289 	{
290 		create: function( settings )
291 		{
292 			var s = $.extend( {}, defaults, settings || {} );
293 			
294 			this.modal = $( "#" + s.modalID );
295 			if( !this.modal.length )
296 			{
297 				this.modal = $( '<' + s.modal + '></' + s.modal + '>' )
298 					.attr( "id", s.modalID )
299 					.appendTo( "body" );
300 			}
301 			
302 			this.modal.show();
303 			
304 			this.prototype.size( this.modal );
305 			//this.zindex();
306 			
307 			return true;	
308 		},
309 		
310 		show: function( settings )
311 		{
312 			var s = $.extend( {}, defaults, settings || {} );
313 			
314 			this.create();
315 			
316 			!$( "p.modalLoader", this.modal ).length
317 				? $( "<p class='modalLoader'>Carregando...</p>" ).appendTo( this.modal )
318 				: $( "p.modalLoader", this.modal ).show();
319 		},
320 		
321 		hide: function()
322 		{
323 			this.modal.hide();	
324 		}
325 	}
326 	
327 	$.modal.prototype = $m.fn;
328 	
329 })( jQuery );
